## §1.1

```python 
from locust import TaskSet, User, between, task


class SearchProducts(TaskSet):
    @task
    def search_men_products(self):
        print("searching men products")

    @task
    def search_kids_products(self):
        print("searching kids products")

class MyUser(User):
    wait_time = between(1, 2)
    tasks = [SearchProducts,]
```

— один тип *user*,
— один тип *task set*,
— обе таски будут выполняться в случайном порядке

```python 
from locust import SequentialTaskSet, User, between, task

  
class SearchProducts(SequentialTaskSet):

    def search_men_products(self):
```

```python 
class SearchProducts(SequentialTaskSet):
    @task
    def search_men_products(self):
        print("searching men products")

    @task
    def search_kids_products(self):
        print("searching kids products")
    
    @task
    def exit_task(self):  # the method name doesn't matter
        self.interrupt()  # the interrupt() method explicit call matters


class ViewCart(SequentialTaskSet):
    @task
    def get_cart_items(self):
        print("get all cart items")

    @task
    def search_cart_item(self):
        print("searching item from cart")
    
    @task
    def exit_task(self):  # the method name doesn't matter
        self.interrupt()  # the interrupt() method explicit call matters

    tasks = {SearchProducts: 4, ViewCart: 1,}
```

— один тип *user*,
— два типа *sequential task set*,
— оба *sequential task set* будут выбираться для выполнения в случайном порядке, 
— первый *sequential task set* будет выбираться в четыре раза чаще второго,
— таски, определённые в каждом выбранном *sequential task set*, будут выполняться строго последовательно

```python 
from locust import SequentialTaskSet, User, between, events, task


# run once before the whole suite
@events.test_start.add_listener
def on_test_start(**kwargs):
    print("----- on_test_start: initiating load testing -----") 

# run once after the whole suite
@events.test_stop.add_listener
def on_test_stop(**kwargs):
    print("----- on_test_stop: finalizing load testing -----") 


class SearchProducts(SequentialTaskSet):
    # run every time before
    # any user starts performing a task set of this ntype
    def on_start(self):
        print("SearchProducts: starting task set")

    # run every time after
    # any user starts performing a task set of this ntype
    def on_stop(self):
        print("SearchProducts: finishing task set") 

    @task
    def search_men_products(self):
        print("searching men products")

    @task
    def search_kids_products(self):
        print("searching kids products")

    @task
    def exit_task(self):
        self.interrupt()


class MyUser(User):
    wait_time = between(1, 2)
    tasks = [SearchProducts, ]

    # run before every user of this type
    # is hatched
    # and start his first task
    def on_start(self):
        print("MyUser: hatchung new user")

    # run after every user of this type
    # complete all his tasks
    # and to be deleted
    def on_stop(self):
        print("MyUser: deleting user") 
        
```
