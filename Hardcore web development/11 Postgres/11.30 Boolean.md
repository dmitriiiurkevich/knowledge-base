
```sql
drop table if exists user, email cascade;

create table user(
    user_id bigint generated always as identity primary key,
    birth_date DATE
);

create table email (
    email varchar(320),
    user_id bigint references user(user_id),
    is_verified boolean not null default false, --> дефолтное значение
    primary key (email, user_id) --> составной первичный ключ
);
```

Составной первичный `(email, user_id)` ключ даёт нам:
- поле `email` не может содержать *null*
- поле `user_id` не может содержать *null*
- все пары значений `(email, user_id)` уникальны


```sql
create unique index unique_verified_email_per_user
on email(email)
where is_verified = true;
```

Индекс гарантирует, что не может быть двух записей с одним и тем же верифицированным имейлом.

Создаём двух пользователей

```sql
insert into user(birth_date) values ('1988-07-29') returning user_id; --> 1
insert into user(birth_date) values ('1996-01-01') returning user_id; --> 2
```

с одинаковыми имейлами

``` sql
insert into email(email, user_id) values ('my@mail.com', 1);
insert into email(email, user_id) values ('my@mail.com', 2);
```


``` sql
update email set is_verified=true where email='my@mail.com' and user_id=1;
-- первая авторизация имейла проходит успешно

update email set is_verified=true where email='my@mail.com' and user_id=2;
-- повторная авторизация имейла вызывает ошибку
-- поскольку срабатывает индекс
```
