
Колонка *is_deleted*

``` sql
# create temp table some_entity(
    entity_id bigint generated always as identity primary key,
    value text not null,
    is_deleted boolean not null default false
);
```

``` sql
# insert into some_entity (value) values ('some value');
# insert into some_entity (value) values ('some another value');
```

Удаление с помощью апдейта

``` sql
# update some_entity set is_deleted=true where entity_id=2;
```

Таблица с историческими данными

``` sql
# drop table if exists some_entity;

# create temp table some_entity(
    entity_id bigint generated always as identity primary key,
    value text not null
);
```

``` sql
# drop table if exists some_entity_deleted;

# create table some_entity_deleted (
	like some_entity including defaults including constraints including indexes
);
```

```sql
# insert into some_entity (value) values ('some value');
# insert into some_entity (value) values ('some another value');
```

Удаление с помощью транзакции

``` sql
# begin;
insert into some_entity_deleted select * from some_entity where entity_id=2;
delete from some_entity where entity_id=2;
commit;
```

Удаление с помощью блока *CTE* — *Common Table Expression*

```sql
# insert into some_entity (value) values ('some another value');

# with deleted_rows as (
    delete from some_entity
    where entity_id=3
    returning *
)
insert into some_entity_deleted select * from deleted_rows;
-- атомарный запрос
```

Два способа создать таблицу аналогичную уже имеющейся

```sql
# create table some_entity_deleted (
	like some_entity 
	including defaults including constraints including indexes
);
```


```sql
# create table some_entity_deleted as select * from some_entity;
```
