
Откуда можно сделать *select* запрос?
- из таблицы
- из результатов другого *select* запроса
- из *view* или из *materialized view*
- из функции (результата вызова функции)
- из *values* или из виртуальной таблицы

Более коротки вариант запроса `select *`

``` SQL
# select * from my_table;

-- То же самое, только короче 
# table my_table;
```


Явная проверка того, что логические операторы языка *SQL* ленивы.

``` SQL
-- Сделаем таблицу, в которую будем логировать вызов функций
# drop table if exists call_log;
# create table call_log (
    id int generated always as identity primary key,
    function_name text,
    call_time timestamp default current_timestamp
);

-- Вызов test_true() логирует своё выполнение и возвращает true
# create or replace function test_true() returns boolean as $$
begin
    insert into call_log (function_name) values ('test_true');
    return true;
end;
$$ language plpgsql;

-- Вызов test_false() логирует своё выполнение и возвращает false
# create or replace function test_false() returns boolean as $$
begin
    insert into call_log (function_name) values ('test_false');
    return false;
end;
$$ language plpgsql;

-- Проверяем, будет ли вызвана test_false()
# select test_true() or test_false();
# select * from call_log;
-- Была вызвана только test_true(), а test_false() не была вызвана
```

Двойные кавычки используются если:  
- в названии колонки используется ключевое слово, например, `"select"`  
- если нужно задать регистр слова, например, не `name`, а `"NaMe"`  
- если в названии колонки будут использованы пробелы, например, `"first name"`

По умолчанию Postgres приводит имена колонок к нижнему регистру (а вот Oracle, например, – к верхнему). Если же нужно сохранить регистр имён, то лучше обернуть их в кавычки:

```sql
select a as A from (values (1), (2)) t(a); 
--> a: 1 2
  
select a as "A" from (values (1), (2)) t(a); 
--> A: 1 2
```


NULL

``` sql
# select 2 > 3; --> true

# select 2 < 3; --> false

# select 2 > null; --> null

# select null < null; --> null

# select null is null; --> true

# select 2 is null; --> false
```

Сортировка

``` SQL
# select * from my_table
order by 1 asc nulls first;


# select * from my_table
order by 1 asc nulls last;
```

В разных СУБД по умолчанию используются разные режимы:
- в SQLite — *nulls first*
- в PostgreSQL — *nulls last*
Это может вызвать проблемы при миграции из одной базы данных в другую.

Сортировка называется устойчивой, если она не меняет взаимного расположения элементов с одинаковыми ключами. Сортировка в SQL неустойчивая.

Ограничение количества

``` SQL
# select * from my_table
limit 10;


# select * from my_table
limit 10 offset 0;

# select * from my_table
limit 10 offset 30; --> 4-я страница по 10 записей
```

Offset работает как пагинация, select со слишком большим значением offset не вернёт ни одной строки. 

Select без сортировки возвращает строки в непредсказуемом порядке. Поэтому select с лимитированием, но без сортировки, возвращает, вообще говоря, непредсказуемый набор строк.

``` sql
# drop table if exists some_values;
# create temp table some_values (value int);
# insert into some_values (value) values (1), (2), (3), (4), (5);

# select * from some_values limit 1; --> 1

# update some_values set value=777 where value = 1;
-- UPDATE может произойти где-то в другой транзакции

# select * from some_values limit 1; --> 2
```

Insert + overriding

если нужно передать результат select в insert, то можно поступит так

``` sql
insert into tests.author (name, description)  
select name, description from public.author;
```

а можно так, если хочется сохранить значения автоматически создаваемых атрибутов

``` sql
insert into tests.author OVERRIDING SYSTEM VALUE select * from public.author;
```

Upsert: Insert + on conflict

``` sql
insert into books_with_isbn (name, isbn) values (
    'My Book', '978-5-389-16579-3')
on conflict(isbn) do update 
set name = excluded.name;
```


``` sql
insert into books_with_isbn (name, isbn) values (
    'My Book', '978-5-389-16579-3') 
on conflict(isbn) do nothing;
```


Insert с возвращаемым значением

``` sql
insert into something (name) ('my value') returning *;

insert into something (name) ('my value') returning something_id;
```


Создание таблицы из результатов запроса

``` sql
create table crows as select * from species where genus_id=(
    select genus_id from genus where genus_name='Ворон'
)
```

или

``` sql
select * into crows from species s where s.genus_id=(
    select genus_id from genus g where g.genus_name='Ворон'
);
```
