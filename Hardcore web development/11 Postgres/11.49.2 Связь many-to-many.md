``` sql
create table book_category (
category_id bigint generated always as identity primary key,
name varchar(150) not null check (length(name) >= 2)
);

create table author (
author_id bigint generated always as identity primary key,
name varchar(150) not null check (length(name) >= 3),
description text check (length(description) >= 30)
);

create table book (
book_id bigint generated always as identity primary key,
created_at timestamp,
name varchar(255) not null check (length(name) >= 2),
description text check (length(description) >= 30),
cover varchar(255)
);

create table book_to_category (
book_id bigint references book(book_id),
category_id bigint references book_category(category_id),
primary key (book_id, category_id)
);

create table book_to_author (
book_id bigint references book(book_id),
author_id bigint references author(author_id),
primary key (book_id, author_id)
);
```

Список книг с категориями

```sql
select b.name book_name, c.name category
from book b
join book_to_category bc using(book_id)
join book_category c using(category_id)
order by book_name, category_id;
```

Список книг с категориями, агрегированными в массив

```sql
select b.name book_name, array_agg(c.name) category
from book b
join book_to_category bc using(book_id)
join book_category c using(category_id)
group by book_name
order by book_name;
```

Список книг с категориями, агрегированными в строку

```sql
select b.name book_name, string_agg(c.name, ', ') category_id
from book b
join book_to_category bc using(book_id)
join book_category c using(category_id)
group by book_name
order by book_name;
```


Обеспечение ссылочной целостности данных  при использовании бридж таблиц

Для обеспечения ссылочной целостности данных на уровне *СУБД* можно создавать специальные триггеры, которые при добавлении или изменении книг будут явно проверять, наличие всех необходимых записей во всех бриджевых таблицах. В случае, если какая-то запись не будет обнаружена, соответствующий триггер выбросит исключение, которое прервёт выполнение транзакции.

1.
Сначала создаём функцию, которая будет вызываться при добавлении или изменении книг.

```sql
CREATE OR REPLACE FUNCTION check_book_has_author() RETURNS TRIGGER AS $$
BEGIN
    IF (SELECT COUNT(*) FROM book_to_author WHERE book_id = NEW.book_id) = 0 THEN
        RAISE EXCEPTION 'No author is found';
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;
```

Эта функция проверяет, что для добавленной или измененной книги есть хотя бы одна запись в связующей таблице `book_to_author`.

2.
Потом создаём триггер, запускающий эту функцию при каждой попытке добавления новой или изменения уже существующей книги.

```sql
CREATE TRIGGER trg_check_book_has_author
AFTER INSERT OR UPDATE ON book
FOR EACH ROW
EXECUTE FUNCTION check_book_has_author();
```

3.
Пробуем создать новую книгу без ссылки на автора.

```sql
insert into book(name) values ('New book name');
-- SQL Error [P0001]: No author is found
```

4.
Чтобы всё получилось, нужно одновременно — в одной транзакции — создать и книгу, и запись в бриджевой таблице.

```sql
WITH inserted_book AS (
    INSERT INTO book (name) VALUES ('New book name')
    RETURNING book_id
)
INSERT INTO book_to_author (book_id, author_id)
SELECT inserted_book.book_id, 1 --> author id
FROM inserted_book;
```

Запрос отрабатывает без ошибки, потому что он выполняется как единое целое.
